1   package org.metaborg.spoofax.shell.commands;
2   
3   import java.io.IOException;
4   import java.util.function.Consumer;
5   
6   import org.apache.commons.vfs2.FileObject;
7   import org.metaborg.core.MetaborgException;
8   import org.metaborg.spoofax.core.analysis.ISpoofaxAnalysisService;
9   import org.metaborg.spoofax.core.analysis.ISpoofaxAnalyzeResult;
10  import org.metaborg.spoofax.core.unit.ISpoofaxAnalyzeUnit;
11  import org.metaborg.spoofax.core.unit.ISpoofaxParseUnit;
12  import org.metaborg.spoofax.shell.core.StyledText;
13  import org.spoofax.interpreter.terms.IStrategoTerm;
14  
15  import com.google.inject.Inject;
16  import com.google.inject.name.Named;
17  
18  /**
19   * Represents an analyze command sent to Spoofax.
20   */
21  public class AnalyzeCommand extends SpoofaxCommand {
22      private static final String DESCRIPTION = "Analyze an expression.";
23  
24      @Inject
25      private ISpoofaxAnalysisService analysisService;
26      @Inject
27      private ParseCommand parseCommand;
28  
29      /**
30       * Instantiate an {@link AnalyzeCommand}.
31       * @param onSuccess Called upon success by the created {@link SpoofaxCommand}.
32       * @param onError   Called upon an error by the created {@link SpoofaxCommand}.
33       */
34      @Inject
35      public AnalyzeCommand(@Named("onSuccess") final Consumer<StyledText> onSuccess,
36                            @Named("onError") final Consumer<StyledText> onError) {
37          super(onSuccess, onError);
38      }
39  
40      @Override
41      public String description() {
42          return DESCRIPTION;
43      }
44  
45      /**
46       * Analyzes a program using the {@link ISpoofaxAnalysisService}.
47       * Delegates parsing to the {@link ParseCommand}.
48       * @param source the source of the program
49       * @param sourceFile the file containing the source of the program
50       * @return an {@link ISpoofaxAnalyzeUnit}
51       * @throws MetaborgException when parsing or analyzing fails
52       * @throws IOException when creating a temp file fails
53       */
54      public ISpoofaxAnalyzeUnit analyze(String source, FileObject sourceFile)
55              throws MetaborgException, IOException {
56          ISpoofaxParseUnit parse = parseCommand.parse(source, sourceFile);
57          ISpoofaxAnalyzeUnit analyze = this.analyze(parse);
58          return analyze;
59      }
60  
61      /**
62       * Analyzes a program using the {@link ISpoofaxAnalysisService}.
63       * @param parseUnit a {@link ParseCommand} result
64       * @return an {@link ISpoofaxAnalyzeUnit}
65       * @throws MetaborgException when analyzing fails
66       */
67      public ISpoofaxAnalyzeUnit analyze(ISpoofaxParseUnit parseUnit) throws MetaborgException {
68          ISpoofaxAnalyzeResult analyzeResult = analysisService.analyze(parseUnit, this.context);
69          ISpoofaxAnalyzeUnit analyzeUnit = analyzeResult.result();
70  
71          StringBuilder builder = new StringBuilder();
72          analyzeUnit.messages().forEach(builder::append);
73  
74          if (!analyzeUnit.valid()) {
75              throw new MetaborgException(builder.toString());
76          }
77          return analyzeUnit;
78      }
79  
80      @Override
81      public void execute(String... args) {
82          try {
83              FileObject tempFile = this.context.location().resolveFile("tmp.src");
84              IStrategoTerm term = this.analyze(args[0], tempFile).ast();
85  
86              this.onSuccess.accept(new StyledText(common.toString(term)));
87          } catch (IOException | MetaborgException e) {
88              this.onError.accept(new StyledText(e.getMessage()));
89          }
90      }
91  }
