3   import java.awt.Color;
4   import java.util.Collections;
5   import java.util.Map;
6   import java.util.Set;
7   import java.util.function.Consumer;
8   
9   import org.apache.commons.lang3.StringUtils;
10  import org.metaborg.spoofax.shell.core.StyledText;
11  
12  import com.google.inject.Inject;
13  import com.google.inject.Provider;
14  import com.google.inject.name.Named;
15  
16  /**
17   * Shows descriptions of all commands, or one command if given.
18   */
19  public class HelpCommand implements IReplCommand {
20      private Map<String, IReplCommand> commands;
21      private Provider<ICommandInvoker> provider;
22      private Consumer<StyledText> successHook;
23      private Consumer<StyledText> errorHook;
24  
25      /**
26       * @param commands
27       *            The commands map, which contains all names too.
28       * @param commandInvokerProvider
29       *            A {@link Provider} for an {@link ICommandInvoker}.
30       * @param successHook
31       *            Called when a given command and its description was found.
32       * @param errorHook
33       *            Called when a given command was not found.
34       */
35      @Inject
36      public HelpCommand(Map<String, IReplCommand> commands,
37                         Provider<ICommandInvoker> commandInvokerProvider,
38                         @Named("onSuccess") Consumer<StyledText> successHook,
39                         @Named("onError") Consumer<StyledText> errorHook) {
40          this.commands = commands;
41          this.provider = commandInvokerProvider;
42          this.successHook = successHook;
43          this.errorHook = errorHook;
44      }
45  
46      @Override
47      public String description() {
48          return "You just used it.";
49      }
50  
51      @Override
52      public void execute(String... args) {
53          StringBuilder output = new StringBuilder();
54          ICommandInvoker invoker = provider.get();
55          Set<String> commandNames = commands.keySet();
56          if (args.length > 0) {
57              String commandName = args[0];
58              commandNames = Collections.singleton(commandName);
59              try {
60                  invoker.commandFromName(commandName);
61              } catch (CommandNotFoundException e) {
62                  errorHook.accept(new StyledText(Color.RED, e.getMessage()));
63              }
64          }
65          int longestCommand = commandNames.stream()
66              .max((a, b) -> Integer.compare(a.length(), b.length())).orElse("").length();
67  
68          // @formatter:off
69          // commandPrefix()commandfoobar commandDescription
70          // commandPrefix()commandfoo    commandDescription
71          commandNames.forEach(name ->
72              output.append(invoker.commandPrefix()).append(name)
73                    .append(StringUtils.repeat(' ', longestCommand - name.length() + 1))
74                    .append(commands.get(name).description()).append('\n'));
75          // @formatter:on
76          successHook.accept(new StyledText(output.toString()));
77      }
78  
79  }
