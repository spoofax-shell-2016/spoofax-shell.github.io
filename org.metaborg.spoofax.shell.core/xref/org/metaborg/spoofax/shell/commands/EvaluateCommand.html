3   import java.io.IOException;
4   import java.util.function.Consumer;
5   
6   import org.apache.commons.vfs2.FileObject;
7   import org.metaborg.core.MetaborgException;
8   import org.metaborg.core.language.FacetContribution;
9   import org.metaborg.spoofax.core.stratego.IStrategoRuntimeService;
10  import org.metaborg.spoofax.core.stratego.StrategoRuntimeFacet;
11  import org.metaborg.spoofax.core.unit.ISpoofaxAnalyzeUnit;
12  import org.metaborg.spoofax.core.unit.ISpoofaxParseUnit;
13  import org.metaborg.spoofax.shell.core.StyledText;
14  import org.spoofax.interpreter.terms.IStrategoTerm;
15  import org.strategoxt.HybridInterpreter;
16  
17  import com.google.inject.Inject;
18  import com.google.inject.name.Named;
19  
20  /**
21   * Represents an evaluate command sent to Spoofax.
22   */
23  public class EvaluateCommand extends SpoofaxCommand {
24      private static final String DESCRIPTION = "Evaluate an expression";
25  
26      @Inject
27      private IStrategoRuntimeService runtimeService;
28      @Inject
29      private AnalyzeCommand analyzeCommand;
30  
31      /**
32       * Instantiate an {@link EvaluateCommand}.
33       * @param onSuccess Called upon success by the created {@link SpoofaxCommand}.
34       * @param onError   Called upon an error by the created {@link SpoofaxCommand}.
35       */
36      @Inject
37      public EvaluateCommand(@Named("onSuccess") final Consumer<StyledText> onSuccess,
38                             @Named("onError") final Consumer<StyledText> onError) {
39          super(onSuccess, onError);
40      }
41  
42      @Override
43      public String description() {
44          return DESCRIPTION;
45      }
46  
47      /**
48       * Interprets a program using a {@link HybridInterpreter}.
49       * Delegates parsing to the {@link ParseCommand} and analyzing to the {@link AnalyzeCommand}.
50       * @param source the source of the program
51       * @param sourceFile the file containing the source of the progra
52       * @return an {@link IStrategoTerm}
53       * @throws MetaborgException when parsing, analyzing or interpreting fails
54       * @throws IOException when creating a temp file fails
55       */
56      public IStrategoTerm interp(String source, FileObject sourceFile)
57              throws IOException, MetaborgException {
58          return this.interp(analyzeCommand.analyze(source, sourceFile));
59      }
60  
61      /**
62       * Interprets a program using a {@link HybridInterpreter}.
63       * Delegates analyzing to the {@link AnalyzeCommand}.
64       * @param parseUnit a {@link ParseCommand} result
65       * @return an {@link IStrategoTerm}
66       * @throws MetaborgException when parsing, analyzing or interpreting fails
67       * @throws IOException when creating a temp file fails
68       */
69      public IStrategoTerm interp(ISpoofaxParseUnit parseUnit) throws IOException, MetaborgException {
70          return this.interp(analyzeCommand.analyze(parseUnit));
71      }
72  
73      /**
74       * Interprets a program using a {@link HybridInterpreter}.
75       * @param analyzeUnit a {@link AnalyzeCommand} result
76       * @return an {@link IStrategoTerm}
77       * @throws MetaborgException when parsing, analyzing or interpreting fails
78       */
79      public IStrategoTerm interp(ISpoofaxAnalyzeUnit analyzeUnit) throws MetaborgException {
80          final FacetContribution<StrategoRuntimeFacet> runContrib = this.context.language()
81                  .facetContribution(StrategoRuntimeFacet.class);
82          if (runContrib == null) {
83              throw new MetaborgException("Cannot retrieve the runtime facet");
84          }
85          final HybridInterpreter interpreter = runtimeService.runtime(
86                  runContrib.contributor, this.context);
87          return common.invoke(interpreter, analyzeUnit.ast(), "runstrat");
88      }
89  
90      /**
91       * Evaluates {@code args}.
92       * @param args The expression(s) to evaluate.
93       */
94      @Override
95      public void execute(String... args) {
96          try {
97              FileObject tempFile = this.context.location().resolveFile("tmp.src");
98              IStrategoTerm term = this.interp(args[0], tempFile);
99  
100             this.onSuccess.accept(new StyledText(common.toString(term)));
101         } catch (IOException | MetaborgException e) {
102             this.onError.accept(new StyledText(e.getMessage()));
103         }
104     }
105 }
