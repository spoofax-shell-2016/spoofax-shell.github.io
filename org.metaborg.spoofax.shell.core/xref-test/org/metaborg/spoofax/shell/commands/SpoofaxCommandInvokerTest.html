12  import com.google.inject.Guice;

14  /**
15   * Test the default implementation of {@link ICommandInvoker}.
16   */
17  public class SpoofaxCommandInvokerTest {
18      private SpoofaxCommandInvoker invoker;
19  
20      /**
21       * Setup the commands, one with description and one without.
22       */
23      @Before
24      public void setUp() {
25          invoker =
26              Guice.createInjector(new CommandTestModule()).getInstance(SpoofaxCommandInvoker.class);
27      }
28  
29      /**
30       * Test correct throwing of {@link CommandNotFoundException}.
31       */
32      @Test
33      public void testCommandNotFound() {
34          try {
35              invoker.commandFromName("does-not-exist");
36              invoker.execute(":does-not-exist");
37              fail("No exceptions thrown, but command does not exist");
38          } catch (CommandNotFoundException e) {
39              assertCommandNameInMessage("does-not-exist", e);
40          }
41      }
42  
43      private static void assertCommandNameInMessage(String commandName, CommandNotFoundException e) {
44          assertTrue("Command name should be present in CommandNotFoundException message",
45                     e.getMessage().contains('\"' + commandName + '\"'));
46      }
47  }
