26  /**
27   * This class launches a console based REPL.
28   *
29   * It uses a GNU Readline-like input buffer with multiline editing capabilities, keyboard shortcuts
30   * and persistent history. ANSI color codes are used to display colors.
31   */
32  public final class ConsoleRepl {
33  
34      private ConsoleRepl() {
35      }
36  
37      private static IProject createProject(final Spoofax spoofax) throws MetaborgException {
38          final ISimpleProjectService projectService =
39              spoofax.injector.getInstance(SimpleProjectService.class);
40          final FileObject projectPath = spoofax.resourceService.resolve("tmp://");
41          return projectService.create(projectPath);
42      }
43  
44      /*
45       * Finds a (the first) language from a parent location. TODO: Eventually, this should find out
46       * which language is actually in use instead of taking the first language found.
47       */
48      private static ILanguageImpl findLanguage(final Spoofax spoofax, final FileObject langPath)
49          throws MetaborgException {
50          final Iterable<ILanguageDiscoveryRequest> requests =
51              spoofax.languageDiscoveryService.request(langPath);
52          final Iterable<ILanguageComponent> components =
53              spoofax.languageDiscoveryService.discover(requests);
54  
55          final Set<ILanguageImpl> implementations = LanguageUtils.toImpls(components);
56          final ILanguageImpl lang = LanguageUtils.active(implementations);
57          if (lang == null) {
58              throw new MetaborgException("Cannot find a language implementation");
59          }
60          return lang;
61      }
62  
63      private static IContext createContext(final Spoofax spoofax, final String langPath)
64          throws MetaborgException {
65          final IProject project = createProject(spoofax);
66          final FileObject zipFile = spoofax.resourceService.resolve(langPath);
67          // TODO: do not hardcode zipfiles
68          final FileObject langFile = spoofax.resourceService.resolve("zip:" + zipFile + "!/");
69          final ILanguageImpl lang = findLanguage(spoofax, langFile);
70          return spoofax.contextService.getTemporary(project.location(), project, lang);
71      }
72  
73      /**
74       * Instantiates and runs a new Repl.
75       *
76       * @param args
77       *            The path to a language implementation location, using any URI supported by Apache
78       *            VFS.
79       * @throws IOException
80       *             When an IO error occurs during execution.
81       */
82      public static void main(String[] args) throws IOException {
83          if (args.length == 0) {
84              throw new IllegalArgumentException("Expected a path to a language "
85                                                 + "implementation location");
86          }
87  
88          try (final Spoofax spoofax = new Spoofax()) {
89              final IContext context = createContext(spoofax, args[0]);
90              final Injector injector = Guice.createInjector(new ConsoleReplModule(context));
91              final IEditor editor = injector.getInstance(IEditor.class);
92              final IDisplay display = injector.getInstance(IDisplay.class);
93              final Repl repl = injector.getInstance(Repl.class);
94              final StyledText message = new StyledText(Color.BLUE, "Welcome to the ")
95                  .append(Color.GREEN, "Spoofax").append(Color.BLUE, " REPL");
96              display.displayResult(message);
97              editor.history().loadFromDisk();
98              repl.run();
99              editor.history().persistToDisk();
100         } catch (IOException | MetaborgException e) {
101             e.printStackTrace();
102         }
103     }
104 }
