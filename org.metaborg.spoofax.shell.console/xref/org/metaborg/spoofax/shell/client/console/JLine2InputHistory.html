3   import java.util.List;
4   import java.util.stream.Collectors;
5   import java.util.stream.StreamSupport;
6   
7   import org.metaborg.spoofax.shell.client.IInputHistory;
8   
9   import com.google.inject.Inject;
10  
11  /**
12   * An adapter for JLine2's History implementation.
13   */
14  public class JLine2InputHistory implements IInputHistory {
15      protected jline.console.ConsoleReader reader;
16      private final jline.console.history.MemoryHistory delegateHist;
17  
18      /**
19       * @param reader
20       *            The {@link jline.console.ConsoleReader ConsoleReader} for setting the history
21       *            being used.
22       * @param delegateHist
23       *            The jline2 history implementation that will be delegated to.
24       */
25      @Inject
26      JLine2InputHistory(final jline.console.ConsoleReader reader,
27                         final jline.console.history.MemoryHistory delegateHist) {
28          this.reader = reader;
29          this.delegateHist = delegateHist;
30          reader.setHistory(delegateHist);
31      }
32  
33      /**
34       * @return The delegate that is currently active. When the
35       *         {@link jline.console.history.FileHistory FileHistory} delegate has not been loaded
36       *         from disk, the {@link jline.console.history.MemoryHistory MemoryHistory} delegate is
37       *         used instead.
38       */
39      protected jline.console.history.History delegate() {
40          return delegateHist;
41      }
42  
43      @Override
44      public String get(int index) {
45          return delegate().get(index).toString();
46      }
47  
48      @Override
49      public void append(String newEntry) {
50          delegate().add(newEntry);
51      }
52  
53      @Override
54      public int size() {
55          return delegate().size();
56      }
57  
58      @Override
59      public List<String> entries(int from, int to) {
60          // @formatter:off
61          return StreamSupport.stream(delegate().spliterator(), false)
62              .skip(from)
63              .limit(to - from)
64              .map(entry -> entry.value().toString())
65              .collect(Collectors.toList());
66          // @formatter:on
67      }
68  }
