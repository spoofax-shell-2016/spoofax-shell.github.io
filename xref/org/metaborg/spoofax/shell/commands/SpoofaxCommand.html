1   package org.metaborg.spoofax.shell.commands;

3   import java.util.function.Consumer;
4   
5   import org.metaborg.core.context.IContext;
6   import org.metaborg.spoofax.core.stratego.IStrategoCommon;
7   import org.metaborg.spoofax.shell.core.StyledText;
8   
9   import com.google.inject.Inject;
10  import com.google.inject.name.Named;
11  
12  /**
13   * Command for processing a String as an expression in some language.
14   */
15  public abstract class SpoofaxCommand implements IReplCommand {
16      private static final String DESCRIPTION = "Process an expression in some language";
17      protected Consumer<StyledText> onSuccess;
18      protected Consumer<StyledText> onError;
19  
20      @Inject
21      protected IStrategoCommon common;
22      @Inject
23      protected IContext context;
24  
25      /**
26       * Instantiate a {@link SpoofaxCommand}.
27       * @param onSuccess Called upon success by the created {@link SpoofaxCommand}.
28       * @param onError   Called upon an error by the created {@link SpoofaxCommand}.
29       */
30      @Inject
31      SpoofaxCommand(@Named("onSuccess") final Consumer<StyledText> onSuccess,
32                     @Named("onError") final Consumer<StyledText> onError) {
33          this.onSuccess = onSuccess;
34          this.onError = onError;
35      }
36  
37      @Override
38      public String description() {
39          return DESCRIPTION;
40      }
41  
42      /**
43       * Executes a command.
44       * @param args the command arguments
45       */
46      @Override
47      public abstract void execute(String... args);
48  }
