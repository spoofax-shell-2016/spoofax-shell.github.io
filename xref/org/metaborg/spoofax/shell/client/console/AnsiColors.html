1   package org.metaborg.spoofax.shell.client.console;

3   import java.awt.Color;
4   import java.util.Arrays;
5   
6   import org.fusesource.jansi.Ansi;
7   
8   /**
9    * Represents a list of possible ansi colors and maps them to Java colors.
10   */
11  public enum AnsiColors {
12      BLACK(Ansi.Color.BLACK, Color.BLACK),
13      RED(Ansi.Color.RED, Color.RED),
14      GREEN(Ansi.Color.GREEN, Color.GREEN),
15      YELLOW(Ansi.Color.YELLOW, Color.YELLOW),
16      BLUE(Ansi.Color.BLUE, Color.BLUE),
17      MAGENTA(Ansi.Color.MAGENTA, Color.MAGENTA),
18      CYAN(Ansi.Color.CYAN, Color.CYAN),
19      WHITE(Ansi.Color.WHITE, Color.WHITE);
20  
21      private Ansi.Color ansiColor;
22      private Color color;
23  
24      /**
25       * Construct all ansi color to color mappings.
26       * @param ansiColor the ansi color
27       * @param color the java color
28       */
29      AnsiColors(Ansi.Color ansiColor, Color color) {
30          this.ansiColor = ansiColor;
31          this.color = color;
32      }
33  
34      /**
35       * Find the ansi color most corresponding to a java color.
36       * @param color the java color
37       * @return the ansi color
38       */
39      public static Ansi.Color findClosest(Color color) {
40          return Arrays.stream(AnsiColors.values())
41              .min((a, b) -> Integer.compare(colorDiff(a.color, color), colorDiff(b.color, color)))
42              .get()
43              .ansiColor;
44      }
45  
46      private static int colorDiff(Color a, Color b) {
47          return Math.abs(a.getBlue() - b.getBlue())
48               + Math.abs(a.getGreen() - b.getGreen())
49               + Math.abs(a.getRed() - b.getRed());
50      }
51  }
