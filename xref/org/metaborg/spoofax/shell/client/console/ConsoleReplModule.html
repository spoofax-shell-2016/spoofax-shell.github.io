3   import java.io.IOException;
4   import java.io.InputStream;
5   import java.io.OutputStream;
6   
7   import org.metaborg.core.context.IContext;
8   import org.metaborg.spoofax.shell.client.IDisplay;
9   import org.metaborg.spoofax.shell.client.IEditor;

12  
13  import com.google.inject.Provides;
14  import com.google.inject.Singleton;
15  import com.google.inject.name.Named;
16  import com.google.inject.name.Names;
17  
18  /**
19   * Bindings for the console REPL.
20   */
21  public class ConsoleReplModule extends ReplModule {
22  
23      /**
24       * Instantiates a new ConsoleReplModule.
25       *
26       * @param context
27       *            The {@link IContext} in which the REPL operates.
28       */
29      public ConsoleReplModule(IContext context) {
30          super(context);
31      }
32  
33      private void configureUserInterface() {
34          bind(JLine2InputHistory.class).to(JLine2PersistentInputHistory.class);
35  
36          bind(TerminalUserInterface.class).in(Singleton.class);
37          bind(IEditor.class).to(TerminalUserInterface.class);
38          bind(IDisplay.class).to(TerminalUserInterface.class);
39  
40          bind(InputStream.class).annotatedWith(Names.named("in")).toInstance(System.in);
41          bind(OutputStream.class).annotatedWith(Names.named("out")).toInstance(System.out);
42          bind(OutputStream.class).annotatedWith(Names.named("err")).toInstance(System.err);
43  
44          bindConstant().annotatedWith(Names.named("historyPath"))
45              .to(System.getProperty("user.home") + "/.spoofax_history");
46      }
47  
48      @Override
49      protected void configure() {
50          super.configure();
51  
52          configureUserInterface();
53  
54          bind(Repl.class).in(Singleton.class);
55      }
56  
57      /**
58       * TODO: Replace with "CheckedProvides" because IO in Provider method is bad practice, see:
59       * https://github.com/google/guice/wiki/ThrowingProviders.
60       *
61       * @param in
62       *            The {@link InputStream}.
63       * @param out
64       *            The {@link OutputStream}.
65       * @return a {@link jline.console.ConsoleReader} with the given streams.
66       * @throws IOException
67       *             When an IO error occurs upon construction.
68       */
69      @Provides
70      @Singleton
71      protected jline.console.ConsoleReader provideConsoleReader(@Named("in") InputStream in,
72                                                                 @Named("out") OutputStream out)
73                                                                     throws IOException {
74          return new jline.console.ConsoleReader(in, out);
75      }
76  }
