14  /**
15   * Test cases for the {@link StyledText} class, which wraps Spoofax {@link RegionStyle}.
16   */
17  public class StyledTextTest {
18  
19      /**
20       * Test a string with no style.
21       */
22      @Test
23      public void testString() {
24          StyledText styledText = new StyledText("Hello world");
25  
26          assertEquals("Hello world", styledText.toString());
27          assertEquals(1, styledText.getSource().size());
28          assertNull(styledText.getSource().get(0).style());
29      }
30  
31      /**
32       * Test a string styled with a color.
33       */
34      @Test
35      public void testColoredString() {
36          StyledText styledText = new StyledText(Color.RED, "Hello ").append("world");
37  
38          assertEquals("Hello world", styledText.toString());
39          assertEquals(2, styledText.getSource().size());
40          assertNotNull(styledText.getSource().get(0).style());
41          assertNull(styledText.getSource().get(1).style());
42      }
43  
44      /**
45       * Test a string styled with an arbitrary style.
46       */
47      @Test
48      public void testStyledString() {
49          Style style1 = new Style(Color.BLUE, Color.CYAN, false, false, false);
50          Style style2 = new Style(Color.RED, Color.YELLOW, false, false, false);
51          StyledText styledText = new StyledText(style1, "Hello ").append(style2, "world");
52  
53          assertEquals("Hello world", styledText.toString());
54          assertEquals(2, styledText.getSource().size());
55          assertEquals(style1, styledText.getSource().get(0).style());
56          assertEquals(style2, styledText.getSource().get(1).style());
57      }
58  }
